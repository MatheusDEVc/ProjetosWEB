PASSO 3: Atualizar cart.js
──────────────────────────

Adicionar validações ao addToCart:

function addToCart(productId, quantity = 1) {
    // ✅ VALIDAR
    if (!SecurityValidator.validateProductId(productId)) {
        console.error('ID inválido');
        return false;
    }
    if (!SecurityValidator.validateQuantity(quantity)) {
        console.error('Quantidade inválida');
        return false;
    }
    
    // ✅ OBTER PREÇO SEGURO
    const price = getPriceFromServer(productId);
    
    // ... resto do código
    
    cart.push({ id: productId, quantity, price });
    saveCart(cart);
    return true;
}
